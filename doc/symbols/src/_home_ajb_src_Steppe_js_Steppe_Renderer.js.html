 10     Steppe.Renderer = function(canvas, undefined) {
 11         var _CANVAS_WIDTH        = 320,	// 320 pixels
 12             _CANVAS_HEIGHT       = 200,	// 200 pixels
 13             _ANGLE_OF_VIEW       = 60,	// 60 degrees
 14             _ONE_DEGREE_ANGLE    = 1 / _ANGLE_OF_VIEW * _CANVAS_WIDTH,
 15             _THIRTY_DEGREE_ANGLE = _ONE_DEGREE_ANGLE * 30,
 16             _THREE_HUNDRED_AND_SIXTY_DEGREE_ANGLE = _ONE_DEGREE_ANGLE * 360,
 17             _ANGULAR_INCREMENT   = _ANGLE_OF_VIEW / _CANVAS_WIDTH,
 18             _DEGREES_TO_RADIANS  = Math.PI / 180,
 19             _FAKE_DEGREES_TO_RADIANS = (2 * Math.PI) /
 20                 ((360 / _ANGLE_OF_VIEW) * _CANVAS_WIDTH),
 21             _RADIANS_TO_DEGREES  = 180 / Math.PI,
 22             _RADIANS_TO_FAKE_DEGREES = ((360 / _ANGLE_OF_VIEW) *
 23                 _CANVAS_WIDTH) / (2 * Math.PI),
 24             _SCALE_FACTOR = 35,
 25             _CAMERA_Y     = 175,
 26             _DISTANCE     = 75,
 27             _WATER_HEIGHT = 64;

 29         var _FASTEST   = 4,
 30             _DONT_CARE = 2,
 31             _NICEST    = 1;

 34             _cosineLookupTable = [],
 35             _framebuffer = undefined,
 36             _heightmap = [],
 37             _inverseDistortionLookupTable = [],
 38             _outOfBoundsHeightmap = [],
 39             _outOfBoundsTexturemap = undefined,
 40             _rayLengthLookupTable = [],
 41             _sineLookupTable = [],
 42             _sky = undefined,
 43             _spriteList = [],
 44             _texturemap = undefined,

 52         /**
 53          * Blend two colours together using an alpha value.
 54          *
 55          * @param {object} firstColor First (or source) colour.
 56          * @param {object} secondColor Second (or destination) colour.
 57          * @param {number} alpha Alpha value in the range 0..255.
 58          * @return {object} Mixed colour.
 59          */
 60         var _alphaBlend = function(firstColor, secondColor, alpha) {
 61             if (alpha < 0) {
 62                 alpha = 0;
 63             } else if (alpha > 255) {
 64                 alpha = 255;
 65             }
 66 
 67             var normalisedAlpha = alpha / 255,
 68                 adjustedAlpha = 1 - normalisedAlpha;
 69 
 70             var mixedRed   = firstColor.red   * normalisedAlpha | 0,
 71                 mixedGreen = firstColor.green * normalisedAlpha | 0,
 72                 mixedBlue  = firstColor.blue  * normalisedAlpha | 0;
 73 
 74             mixedRed   += Math.floor(secondColor.red   * adjustedAlpha);
 75             mixedGreen += Math.floor(secondColor.green * adjustedAlpha);
 76             mixedBlue  += Math.floor(secondColor.blue  * adjustedAlpha);
 77 
 78             return { red: mixedRed, green: mixedGreen, blue: mixedBlue };
 79         };

 81         /**
 82          * Get a pixel from the out-of-bounds texturemap.
 83          *
 84          * @param {number} x The x-coordinate; must be in the range
 85          *                   0..out-of-bounds-texturemap-width - 1.
 86          * @param {number} y The y-coordinate; must be in the range
 87          *                   0..out-of-bounds-texturemap-height - 1.
 88          * @return {object} An object composed of RGB components for the
 89          *                  corresponding pixel.
 90          */
 91         var _getPixelFromOutOfBoundsTexturemap = function(x, y) {
 92             if (_outOfBoundsTexturemap !== undefined) {
 93                 var index = (y << 12) + (x << 2);
 94 
 95                 return {
 96                     red:   _outOfBoundsTexturemap[index],
 97                     green: _outOfBoundsTexturemap[index + 1],
 98                     blue:  _outOfBoundsTexturemap[index + 2]
 99                 };
100             } else {
101                 return {
102                     red:   127,
103                     green: 127,
104                     blue:  127
105                 };
106             }
107         };

109         /**
110          * Get a pixel from the sky canvas.
111          *
112          * @param {number} x The x-coordinate; should be in the range
113          *                   0..sky-width - 1.
114          * @param {number} y The y-coordinate; should be in the range
115          *                   0..sky-height - 1.
116          * @return {object} An object composed of RGB components for the
117          *                  corresponding pixel.
118          */
119         var _getPixelFromSky = function(x, y) {
120             var currentAngle = _camera.angle - _THIRTY_DEGREE_ANGLE;
121 
122             if (currentAngle < 0) {
123                 currentAngle += _THREE_HUNDRED_AND_SIXTY_DEGREE_ANGLE;
124             }
125 
126             if (y < 0) {
127                 y = 0;
128             } else if (y >= 100) {
129                 y = 100 - 1;
130             }
131 
132             var data = _sky.getContext('2d').getImageData(
133                 (currentAngle + x | 0) % 1920, y, 1, 1).data;
134 
135             return { red: data[0], green: data[1], blue: data[2] };
136         };

158         /**
159          * Get the row at which a sprite should be rendered. This is a private
160          * helper method.
161          *
162          * @param {number} x ...
163          * @param {number} z ...
164          * @param {number} ray ...
165          * @return {number} ...
166          */
167         var _getRow = function(x, z, ray) {
168             var cameraVectorX = Math.cos(_camera.angle * _ANGULAR_INCREMENT *
169                 _DEGREES_TO_RADIANS);
170             var cameraVectorZ = Math.sin(_camera.angle * _ANGULAR_INCREMENT *
171                 _DEGREES_TO_RADIANS);
172 
173             var spriteVectorX = x - _camera.x;
174             var spriteVectorZ = z - _camera.z;
175 
176             var vectorLength = Math.sqrt(spriteVectorX * spriteVectorX +
177                 spriteVectorZ * spriteVectorZ);
178 
179             var newVectorLength = vectorLength /
180                 _inverseDistortionLookupTable[ray];
181 
182             var y = Math.round(_DISTANCE * _camera.y / newVectorLength);
183             var row = y + _CANVAS_HEIGHT - 1 - _camera.y;
184 
185             return row;
186         };

188         /**
189          * Initialise the inverse distortion lookup table (for removing
190          * fisheye).
191          */
192         var _initInverseDistortionLookupTable = function() {
193             for (var angleOfRotation = 0;
194                 angleOfRotation < _THIRTY_DEGREE_ANGLE;
195                 ++angleOfRotation) {
196                 var angleOfRotationInRadians = angleOfRotation *
197                     _ANGULAR_INCREMENT * _DEGREES_TO_RADIANS;
198 
199                 _inverseDistortionLookupTable[angleOfRotation +
200                     _THIRTY_DEGREE_ANGLE] = 1 /
201                     Math.cos(angleOfRotationInRadians);
202 
203                 var cosine = Math.cos(angleOfRotationInRadians);
204                 if (cosine != 0) {
205                     _inverseDistortionLookupTable[
206                         _THIRTY_DEGREE_ANGLE - angleOfRotation] = 1 / cosine;
207                 }
208             }
209 
210             _inverseDistortionLookupTable[0]   = 2;
211             _inverseDistortionLookupTable[160] = 1;
212         };
213 
214         /**
215          * Initialise (or recalculate) the ray-length lookup table.
216          *
217          * @param {number} y The position of the camera on the y-axis.
218          * @param {number} distance The distance from the camera to the
219          *                          projection plane.
220          */
221         var _initRayLengthLookupTable = function(y, distance) {
222             for (var ray = 1; ray < _CANVAS_WIDTH; ++ray) {
223                 for (var row = 0; row < _CANVAS_HEIGHT + _CANVAS_HEIGHT / 2;
224                     ++row) {
225                     var invertedRow = _CANVAS_HEIGHT - 1 - row;
226 
227                     var rayLength = _inverseDistortionLookupTable[ray] *
228                         ((distance * y) / (y - invertedRow));
229 
230                     _rayLengthLookupTable[row * _CANVAS_WIDTH + ray] =
231                         rayLength;
232                 }
233             }
234         };
235 
236         /**
237          * Initialise the sine and cosine lookup tables.
238          */
239         var _initSineAndCosineLookupTables = function() {
240             for (var angleOfRotation = 0;
241                 angleOfRotation < _THREE_HUNDRED_AND_SIXTY_DEGREE_ANGLE;
242                 ++angleOfRotation) {
243                 var angleOfRotationInRadians = angleOfRotation *
244                     _ANGULAR_INCREMENT * _DEGREES_TO_RADIANS;
245 
246                 _sineLookupTable[angleOfRotation]   = Math.sin(
247                     angleOfRotationInRadians);
248                 _cosineLookupTable[angleOfRotation] = Math.cos(
249                     angleOfRotationInRadians);
250             }
251         };

253         /**
254          * Render the 360-degree panoramic sky based on the camera's angle
255          * of rotation.
256          */
257         var _renderSky = function() {
258             var angleOfRotation = _camera.angle - _THIRTY_DEGREE_ANGLE;
259 
260             if (angleOfRotation < 0) {
261                 angleOfRotation += _THREE_HUNDRED_AND_SIXTY_DEGREE_ANGLE;
262             }
263 
264             angleOfRotation |= 0; 
265 
266             var skyWidth  = _sky.width;
267             var skyHeight = _CANVAS_HEIGHT / 2 - (_camera.y - _CANVAS_HEIGHT);
268 
269             if (skyHeight > _sky.height) {
270                 skyHeight = _sky.height;
271             }
272 
273             var sy = _camera.y - _CANVAS_HEIGHT;
274 
275             if (sy < 0) {
276                 sy = 0;
277             }
278 
279             if (angleOfRotation + 320 <= skyWidth) {
280                 _framebuffer.drawImage(_sky, angleOfRotation, sy, 320,
281                     skyHeight, 0, 0, 320, skyHeight);
282             } else {
283                 _framebuffer.drawImage(_sky, angleOfRotation, sy,
284                     skyWidth - angleOfRotation, skyHeight, 0, 0,
285                     skyWidth - angleOfRotation, skyHeight);
286                 _framebuffer.drawImage(_sky, 0, sy,
287                     320 - (skyWidth - angleOfRotation),
288                     skyHeight, skyWidth - angleOfRotation, 0,
289                     320 - (skyWidth - angleOfRotation), skyHeight);
290             }
291         };

293         /**
294          * Render the terrain (landscape).
295          */
296         var _renderTerrain = function() {
297             _framebuffer.fillStyle = '#7f7f7f';
298             _framebuffer.fillRect(0, 100, 320, 25);
299 
300             var initialAngle = _camera.angle - _THIRTY_DEGREE_ANGLE;
301 
302             if (initialAngle < 0) {
303                 initialAngle += _THREE_HUNDRED_AND_SIXTY_DEGREE_ANGLE;
304             }
305 
306             initialAngle |= 0;
307 
308             var currentAngle = initialAngle;
309 
310             for (var ray = _quality; ray < _CANVAS_WIDTH; ray += _quality) {
311                 var previousTop = _CANVAS_HEIGHT + _CANVAS_HEIGHT / 2 - 1;
312 
313                 for (var row = _CANVAS_HEIGHT + _CANVAS_HEIGHT / 2 - 1;
314                     row >= 0; --row) {
315                     var rayLength = _rayLengthLookupTable[
316                         (row << 8) + (row << 6) + ray];
317 
318                     var rayX = _camera.x + rayLength *
319                         _cosineLookupTable[currentAngle] | 0;
320                     var rayZ = _camera.z + rayLength *
321                         _sineLookupTable[currentAngle] | 0;
322 
323                     var u = rayX & 1023;
324                     var v = rayZ & 1023;
325 
326                     var height;
327                     if ((rayX < 1024 || rayX > 1024 + 1024 ||
328                         rayZ < 1024 || rayZ > 1024 + 1024) &&
329                         _outOfBoundsHeightmap.length > 0) {
330                         height = _outOfBoundsHeightmap[(v << 10) + u];
331                     } else {
332                         height = _heightmap[(v << 10) + u];
333                     }
334 
335                     var scale = height * _SCALE_FACTOR / (rayLength + 1) | 0;
336 
337                     var top = _CANVAS_HEIGHT / 2 + row - scale,
338                         bottom = top + scale;
339 
340                     if (top < previousTop) {
341                         bottom = previousTop;
342                         previousTop = top;
343 
344                         var color = '';
345 
346                         if (rayX < 1024 || rayX > 1024 + 1024 ||
347                             rayZ < 1024 || rayZ > 1024 + 1024) {
348                             var texel = _getPixelFromOutOfBoundsTexturemap(
349                                 u, v);
350 
351                             if (_fog) {
352                                 var foggedTexel = _alphaBlend(texel,
353                                     { red: 127, green: 127, blue: 127 },
354                                     ~~(row / 100 * 255));
355 
356                                 texel = foggedTexel;
357                             }
358 
359                             color = 'rgb(' + texel.red +
360                                 ', ' + texel.green + ', ' +
361                                 texel.blue + ')';
362                         } else {
363                             if (height < _waterHeight) {
364                                 var data = _getPixelFromSky(ray,
365                                     200 - top);
366 
367                                 var texel = _getPixelFromTexturemap(u, v);
368 
369                                 var mixedColor = _alphaBlend(data,
370                                     texel, ~~((_waterHeight - height) /
371                                     _waterHeight * 255 * 2));
372 
373                                 texel = mixedColor;
374 
375                                 if (_fog) {
376                                     var foggedTexel = _alphaBlend(
377                                         mixedColor,
378                                         { red: 127, green: 127, blue: 127 },
379                                         ~~(row / 100 * 255));
380 
381                                     texel = foggedTexel;
382                                 }
383 
384                                 height = _waterHeight;
385 
386                                 color = 'rgb(' + texel.red + ', ' +
387                                     texel.green + ', ' +
388                                     texel.blue + ')';
389                             } else {
390                                 var texel = _getPixelFromTexturemap(u, v);
391 
392                                 if (_fog) {
393                                     var foggedTexel = _alphaBlend(texel,
394                                         { red: 127, green: 127, blue: 127 },
395                                         ~~(row / 100 * 255));
396 
397                                     texel = foggedTexel;
398                                 }
399 
400                                 color = 'rgb(' + texel.red +
401                                     ', ' + texel.green + ', ' +
402                                     texel.blue + ')';
403                             }
404                         }
405 
406                         // Render sliver...
407                         if (bottom > 199) {
408                             bottom = 199;
409                         }
410 
411                         _framebuffer.fillStyle = color;
412                         if (ray > _quality) {
413                             // Not the left-most ray...
414                             _framebuffer.fillRect(ray, top - _smooth,
415                                 _quality, bottom - top + 1);
416                         } else {
417                             // Left-most ray: we don't cast rays for
418                             // column 0!
419                             _framebuffer.fillRect(0, top - _smooth,
420                                 _quality << 1, bottom - top + 1);
421                         }
422                     }
423                 }
424 
425                 currentAngle += _quality;
426                 if (currentAngle >=
427                     _THREE_HUNDRED_AND_SIXTY_DEGREE_ANGLE) {
428                     currentAngle = 0;
429                 }
430             }
431         };

433         /**
434          * Render the terrain (landscape) with sprites.
435          */
436         var _renderTerrainWithSprites = function() {
437             /*
438              * 1. Construct a list of visible sprites; sprites are visible
439              *    where they fall within the -30..30 (60-degree) horizontal
440              *    field-of-view based on the direction that the camera is
441              *    pointing in.
442              * 2. For each visible sprite, determine its projected 2D coords
443              *    (x and y) and its scaled width and height based on distance
444              *    from the camera. The y coord corresponds to the bottom of the
445              *    sprite and the x coord is the centre of the width of the
446              *    sprite.
447              * 3. After drawing each row of terrain, draw any sprites where the
448              *    sprite's 'projected' row equals the current row. Remove the
449              *    sprite from the list of visible sprites.
450              * 4. Rinse and repeat.
451              */
452 
453             // Fill the upper region of the framebuffer with the fog-colour.
454             _framebuffer.fillStyle = '#7f7f7f';
455             _framebuffer.fillRect(0, 100, 320, 25);
456 
457             // Empty the list of visible sprites.
458             _visibleSpriteList.length = 0;
459 
460             // Calculate the unit vector of the camera.
461             var cameraVectorX = Math.cos(_camera.angle * _ANGULAR_INCREMENT *
462                 _DEGREES_TO_RADIANS);
463             var cameraVectorZ = Math.sin(_camera.angle * _ANGULAR_INCREMENT *
464                 _DEGREES_TO_RADIANS);
465 
466             // For each sprite...
467             for (var i = 0; i < _spriteList.length; ++i) {
468                 var sprite = _spriteList[i];
469 
470                 // Calculate the vector of the sprite.
471                 var spriteVectorX = sprite.x - _camera.x;
472                 var spriteVectorZ = sprite.z - _camera.z;
473 
474                 // Calculate the magnitude (length of the vector) to determine
475                 // the distance from the camera to the sprite.
476                 var vectorLength = Math.sqrt(
477                     spriteVectorX * spriteVectorX +
478                     spriteVectorZ * spriteVectorZ);
479 
480                 // Normalise the sprite vector to become the unit vector.
481                 spriteVectorX /= vectorLength;
482                 spriteVectorZ /= vectorLength;
483 
484                 // Calculate the dot product of the camera and sprite vectors.
485                 var dotProduct = cameraVectorX * spriteVectorX +
486                     cameraVectorZ * spriteVectorZ;
487 
488                 // If the dot product is negative...
489                 if (dotProduct < 0) {
490                     // The sprite is behind the camera, so clearly not in view.
491                     // Move to the next sprite.
492                     continue;
493                 }
494 
495                 // Calculate the angle (theta) between the camera vector and
496                 // the sprite vector.
497                 var theta = Math.acos(dotProduct);
498 
499                 // If the angle (theta) is less than or equal to 30 degrees...
500                 // NOTE: We do NOT need to check the lower bound (-30 degrees)
501                 // because theta will /never/ be negative.
502                 if (theta <= _THIRTY_DEGREE_ANGLE * _FAKE_DEGREES_TO_RADIANS) {
503                     var scale = _SCALE_FACTOR / (vectorLength + 1);
504 
505                     // Scale the projected sprite.
506                     var width  = scale * sprite.image.width  | 0;
507                     var height = scale * sprite.image.height | 0;
508 
509                     // Calculate the cross product. The cross product differs
510                     // from the dot product in a crucial way: the result is
511                     // *signed*!
512                     var crossProduct = cameraVectorX * spriteVectorZ -
513                         spriteVectorX * cameraVectorZ;
514 
515                     // Calculate the projected x coord relative to the
516                     // horizonal centre of the canvas. We add or subtract the
517                     // value dependent on the sign of the cross product.
518                     var x;
519                     if (crossProduct < 0) {
520                         x = _CANVAS_WIDTH / 2 -
521                             theta * _RADIANS_TO_FAKE_DEGREES | 0;
522                     } else {
523                         x = _CANVAS_WIDTH / 2 +
524                             theta * _RADIANS_TO_FAKE_DEGREES | 0;
525                     }
526 
527                     // Calculate the 3D coords of the sprite.
528                     var spriteX = sprite.x;
529                     var spriteY = _heightmap[((sprite.z & 1023) << 10) +
530                         (sprite.x & 1023)];
531                     var spriteZ = sprite.z;
532 
533                     var row = _getRow(spriteX, spriteZ, x);
534 
535                     // Centre the scaled sprite.
536                     x -= width / 2;
537 
538                     var rayX = spriteX;
539                     var rayZ = spriteZ;
540 
541                     var u = rayX & 1023;
542                     var v = rayZ & 1023;
543 
544                     var projectedHeight;
545                     if ((rayX < 1024 || rayX > 1024 + 1024 ||
546                         rayZ < 1024 || rayZ > 1024 + 1024) &&
547                         _outOfBoundsHeightmap.length > 0) {
548                         projectedHeight = _outOfBoundsHeightmap[(v << 10) + u];
549                     } else {
550                         projectedHeight = _heightmap[(v << 10) + u];
551                     }
552 
553                     var projectedScale = projectedHeight * scale;
554 
555                     var top = _CANVAS_HEIGHT / 2 -
556                         (_camera.y - _CANVAS_HEIGHT) + row -
557                         projectedScale,
558                         bottom = top + projectedScale;
559 
560                     // Add the projected sprite to the list of visible sprites.
561                     // NOTE: Thus far, only the scaled width and height,
562                     // vectorLength, x and image are guaranteed to be correct.
563                     _visibleSpriteList.push({
564                         height:       height,
565                         image:        sprite.image,
566                         row:          bottom | 0,
567                         vectorLength: vectorLength,
568                         width:        width,
569                         x:            x | 0,
570                         y:            top - height | 0
571                     });
572                 }
573             }
574 
575             var initialAngle = _camera.angle - _THIRTY_DEGREE_ANGLE;
576 
577             if (initialAngle < 0) {
578                 initialAngle += _THREE_HUNDRED_AND_SIXTY_DEGREE_ANGLE;
579             }
580 
581             initialAngle |= 0;
582 
583             var currentAngle = initialAngle;
584 
585             var maximumRow = _CANVAS_HEIGHT + _CANVAS_HEIGHT / 2 - 1;
586 
587             for (var row = 0; row <= maximumRow; ++row) {
588 
589                 for (var ray = _quality; ray < _CANVAS_WIDTH; ray += _quality) {
590                     var rayLength = _rayLengthLookupTable[
591                         (row << 8) + (row << 6) + ray];
592 
593                     var rayX = _camera.x + rayLength *
594                         _cosineLookupTable[currentAngle] | 0;
595                     var rayZ = _camera.z + rayLength *
596                         _sineLookupTable[currentAngle] | 0;
597 
598                     var u = rayX & 1023;
599                     var v = rayZ & 1023;
600 
601                     var height;
602                     if ((rayX < 1024 || rayX > 1024 + 1024 ||
603                         rayZ < 1024 || rayZ > 1024 + 1024) &&
604                         _outOfBoundsHeightmap.length > 0) {
605                         height = _outOfBoundsHeightmap[(v << 10) + u];
606                     } else {
607                         height = _heightmap[(v << 10) + u];
608                     }
609 
610                     var scale = height * _SCALE_FACTOR / (rayLength + 1) | 0;
611 
612                     var top = _CANVAS_HEIGHT / 2 -
613                         (_camera.y - _CANVAS_HEIGHT) + row - scale,
614                         bottom = top + scale;
615 
616                     var color = '';
617 
618                     if (rayX < 1024 || rayX > 1024 + 1024 ||
619                         rayZ < 1024 || rayZ > 1024 + 1024) {
620                         var texel = _getPixelFromOutOfBoundsTexturemap(
621                             u, v);
622 
623                         if (_fog) {
624                             var foggedTexel = _alphaBlend(texel,
625                                 { red: 127, green: 127, blue: 127 },
626                                 ~~(row / 100 * 255));
627 
628                             texel = foggedTexel;
629                         }
630 
631                         color = 'rgb(' + texel.red +
632                             ', ' + texel.green + ', ' +
633                             texel.blue + ')';
634                     } else {
635                         if (height < _waterHeight) {
636                             var data = _getPixelFromSky(ray,
637                                 200 - top);
638 
639                             var texel = _getPixelFromTexturemap(u, v);
640 
641                             var mixedColor = _alphaBlend(data,
642                                 texel, ~~((_waterHeight - height) /
643                                 _waterHeight * 255 * 2));
644 
645                             texel = mixedColor;
646 
647                             if (_fog) {
648                                 var foggedTexel = _alphaBlend(
649                                     mixedColor,
650                                     { red: 127, green: 127, blue: 127 },
651                                     ~~(row / 100 * 255));
652 
653                                 texel = foggedTexel;
654                             }
655 
656                             height = _waterHeight;
657 
658                             color = 'rgb(' + texel.red + ', ' +
659                                 texel.green + ', ' +
660                                 texel.blue + ')';
661                         } else {
662                             var texel = _getPixelFromTexturemap(u, v);
663 
664                             if (_fog) {
665                                 var foggedTexel = _alphaBlend(texel,
666                                     { red: 127, green: 127, blue: 127 },
667                                     ~~(row / 100 * 255));
668 
669                                 texel = foggedTexel;
670                             }
671 
672                             color = 'rgb(' + texel.red +
673                                 ', ' + texel.green + ', ' +
674                                 texel.blue + ')';
675                         }
676                     }
677 
678                     // Render sliver...
679                     if (bottom > 199) {
680                         bottom = 199;
681                     }
682 
683                     _framebuffer.fillStyle = color;
684                     if (ray > _quality) {
685                         // Not the left-most ray...
686                         _framebuffer.fillRect(ray, top - _smooth,
687                             _quality, bottom - top + 1);
688                     } else {
689                         // Left-most ray: we don't cast rays for
690                         // column 0!
691                         _framebuffer.fillRect(0, top - _smooth,
692                             _quality << 1, bottom - top + 1);
693                     }
694 
695                     currentAngle += _quality;
696                     if (currentAngle >=
697                         _THREE_HUNDRED_AND_SIXTY_DEGREE_ANGLE) {
698                         currentAngle = 0;
699                     }
700                 }
701 
702                 // For each 'visible' sprite...
703                 for (var i = 0; i < _visibleSpriteList.length; ++i) {
704                     // If the current sprite has been removed...
705                     if (_visibleSpriteList[i] === undefined) {
706                         // Move to the next sprite.
707                         continue;
708                     }
709 
710                     var sprite = _visibleSpriteList[i];
711 
712                     // If the current row matches the base of the sprite...
713                     if (row == sprite.row) {
714                         // Draw the sprite.
715                         _framebuffer.drawImage(
716                             sprite.image,
717                             sprite.x,
718                             sprite.y - _smooth,
719                             sprite.width,
720                             sprite.height);
721 
722                         // Remove the sprite from the list of visible sprites.
723                         _visibleSpriteList[i] = undefined;
724                     }
725                 }
726 
727                 // Reset the current angle to -30 degrees of centre.
728                 currentAngle = initialAngle;
729             }
730         };
731 
732         if (arguments.length > 1) {
733             throw('Too many arguments passed to constructor');
734         }
735 
736         if (canvas.width !== _CANVAS_WIDTH) {
737             throw('Canvas width not equal to ' + _CANVAS_WIDTH);
738         }
739         if (canvas.height !== _CANVAS_HEIGHT) {
740             throw('Canvas height not equal to ' + _CANVAS_HEIGHT);
741         }
742 
743         _framebuffer = canvas.getContext('2d');
744 
745         _initSineAndCosineLookupTables();
746         _initInverseDistortionLookupTable();
747         _initRayLengthLookupTable(_CAMERA_Y, _DISTANCE);

756              * @param {number} z The z-coordinate in world space.

761                 var v = z & 1023;
762

828                     angle: _camera.angle /
829                         _THREE_HUNDRED_AND_SIXTY_DEGREE_ANGLE * 360 +

866                 } else if (capability === 'reflection-map') {

888              *
889              * @param {object} camera The object representing the current
